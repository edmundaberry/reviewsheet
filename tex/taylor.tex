\section{Taylor series}

\resizebox{\columnwidth}{!}{
\begin{tabular}{c|c|c}
  \hline\hline
  Function & Series & Example \\
  \hline\hline
  $\cos(x)$ & $\sum\limits_{i=0}^{\infty}(-1)^{i}\frac{x^{2i}}{(2i)!}$ & $1 -\frac{x^2}{2!} + \frac{x^4}{4!} - \cdots$ \\
  \hline
  $\sin(x)$ & $\sum\limits_{i=0}^{\infty}(-1)^{i}\frac{x^{2i+1}}{(2i+1)!}$ & $x -\frac{x^3}{3!} + \frac{x^5}{5!} - \cdots$ \\
  \hline
  $e^x$ & $\sum\limits_{i=0}^{\infty}\frac{x^{i}}{i!}$ & $1 + x + \frac{x^2}{2!} + \cdots$ \\
  \hline
  $\frac{1}{1-x}$ & $\sum\limits_{i=0}^{\infty}x^{i}$  & $1 + x + x^2 + \cdots$ \\
  \hline
  $\arctan(x)$ &  $\sum\limits_{i=0}^{\infty}(-1)^{i}\frac{x^{2i+1}}{2i+1}$ & $x -\frac{x^3}{3} + \frac{x^5}{5} - \cdots$ \\
  \hline
  $\ln(1+x)$ & $\sum\limits_{i=0}^{\infty}(-1)^{i+1}\frac{x^{i}}{i}$ & $x - \frac{x^2}{2} + \frac{x^3}{3} - \cdots$ \\
  \hline\hline
\end{tabular}}
