\section{Identities of trigonometric functions}

\resizebox{\columnwidth}{!}{
\begin{tabular}{ccc}
\hline\hline
$\sin(u \pm v)$ & $=$ & $\sin(u)\cdot\cos(v) \pm sin(u)\cdot\cos(v)$ \\
\hline
$\cos(u \pm v)$ & $=$ & $\sin(u)\cdot\sin(v) \mp cos(u)\cdot\cos(v)$ \\
\hline
$\sin(2u)$ & $=$ & $2\sin(u)\cos(u)$ \\
\hline
& $=$ & $\cos^{2}(u) - \sin^{2}(u)$ \\ 
$\cos(2u)$ & $=$ & $2\cos^2(u) - 1$ \\
& $=$ & $1- 2\sin^2(u)$ \\
\hline
$\sin^{2}(u)$ & $=$ & $\frac{1-\cos(2u)}{2}$ \\
\hline
$\cos^{2}(u)$ & $=$ & $\frac{1+\cos(2u)}{2}$ \\
\hline
$\tan^{2}(u)$ & $=$ & $\sec^2(u) - 1$ \\
\hline
$\cot^{2}(u)$ & $=$ & $\csc^2(u) - 1$ \\
\hline\hline
$e^{i\theta}$ & $=$ & $\cos(\theta) + i\cdot\sin(\theta)$ \\
\hline
$e^{i\pi}$ & $=$ & $-1$ \\
\hline
$e^{i\pi/2}$ & $=$ & $i$ \\
\hline 
$\ln(i)$ & $=$ & $\ln(e^{i\pi/2}) = e^{i\pi/2}$ \\
\hline\hline
\end{tabular}}
